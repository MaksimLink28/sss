from bs4 import BeautifulSoup
import requests
#у змінній dollar ми вказуємо посилання на сайт який будемо парсити
dollar = "https://www.google.com/search?q=%D0%BA%D1%83%D1%80%D1%81%20%D0%B4%D0%BE%D0%BB%D0%BB%D0%B0%D1%80%D0%B0"
#тут ми вкаазуємо user-agent (його можна дізнатися написавши в google: My user agent)
user = {"User-Agent": "Your user agent"}
#у змінній plus ми вказуємо посилання(із змінної dollar) та у змінній headers ми вказуємо наш user-agent(який ми получили із змінної user)
plus = requests.get(dollar, headers = user)
#у змінніх soup ми вказуємо що парсимо іменно html код
soup = BeautifulSoup(plus.content, "html.parser")

Далі нам знадобиться вілкрити сайт який ми будемо парсити (у мому випадку сайт з курсом доллара) і переглянути код сторінки

Для того щоб переглянути код сторінки вам знадобиться

  • перейти на потрібний для вас сайт
  • навести мишку на потрібну для вас інформацію
  • нажати праву кнопку миші ті вибрати пункт "Перевірити"

import requests
from bs4 import BeautifulSoup

dollar = "https://www.google.com/search?q=dollar"
user = {"user-Agent": "Your user agent"}

plus = requests.get(dollar, headers = user)
soup = BeautifulSoup(plus.content, "html.parser")
#тут ми вказуємо яку імено строку кода ми парсимо(у перших душках ми пишемо яку основу ми парсимо(у мому випадку це span) у другому пишемо параметри шо там вказується)
convert = soup.findAll("span",{"class": "DFlfde SwHCTb" ,"data-precision": "2"})
#Тепер ми виводимо все у консоль
print(convert[0].text)
